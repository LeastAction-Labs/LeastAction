""" Operator= import boto3
import time
from utils.logger import log_info, log_error

def initialize(connection, taskId="NA", sessionId="NA"):
    try:
        log_info("task", taskId, "initialize", "start", "Starting SSM client initialization",session_id=sessionId)
        region = connection.get("region", "us-east-1")
        client = boto3.client('ssm', region_name=region)
        log_info("task", taskId, "initialize", "client_created", f"SSM client created for region {region}",session_id=sessionId)
        return client, taskId, sessionId
    except Exception as e:
        log_error("task", taskId, "initialize", "error", f"Failed to initialize SSM client: {str(e)}",session_id=sessionId)
        raise

def run(connection, payload, client, taskId="NA", sessionId="NA"):
    try:
        log_info("task", taskId, "run", "start", "Starting command execution",session_id=sessionId)
        connection_copy = connection.copy()
        connection_copy["Parameters"] = connection_copy.get("Parameters", {}).copy()
        connection_copy["Parameters"]["commands"] = payload if isinstance(payload, list) else [payload]
        
        params = {
            'InstanceIds': connection_copy["InstanceIds"],
            'DocumentName': connection_copy.get("DocumentName", "AWS-RunShellScript"),
            'Parameters': connection_copy["Parameters"]
        }
        
        if connection_copy.get("TimeoutSeconds"):
            params['TimeoutSeconds'] = int(connection_copy["TimeoutSeconds"])
            
        log_info("task", taskId, "run", "sending_command", f"Sending command to instances: {params['InstanceIds']}",session_id=sessionId)
        response = client.send_command(**params)
        
        runDetails = {
            'CommandId': response['Command']['CommandId'],
            'Status': response['Command']['Status'],
            'InstanceIds': response['Command']['InstanceIds'],
            'DocumentName': response['Command']['DocumentName'],
            'execution_type': 'async'

        }
        
        log_info("task", taskId, "run", "command_sent", f"Command sent with ID: {runDetails['CommandId']}",session_id=sessionId)
        return runDetails
        
    except Exception as e:
        log_error("task", taskId, "run", "error", f"Failed to send command: {str(e)}",session_id=sessionId)
        raise

def checkCompletion(client, runDetails, taskId="NA", sessionId="NA"):
    try:
        log_info("task", taskId, "checkCompletion", "start", f"Checking completion for CommandId: {runDetails['CommandId']}",session_id=sessionId)
        command_id = runDetails['CommandId']
        max_retries = 25
        retry_delay = 2
        
        for attempt in range(max_retries):
            try:
                response = client.list_command_invocations(CommandId=command_id)
                
                if not response.get('CommandInvocations'):
                    if attempt < max_retries - 1:
                        log_info("task", taskId, "checkCompletion", "retry", f"No invocations found, retrying in {retry_delay}s (attempt {attempt + 1})",session_id=sessionId)
                        time.sleep(retry_delay)
                        continue
                    else:
                        log_error("task", taskId, "checkCompletion", "no_invocations", "No command invocations found after retries",session_id=sessionId)
                        return {'status': 'Failed', 'error': 'No command invocations found'}
                
                invocation = response['CommandInvocations'][0]
                status = invocation['Status']
                
                completionDetails = {
                    'status': status,
                    'StandardOutputContent': invocation.get('StandardOutputContent', ''),
                    'StandardErrorContent': invocation.get('StandardErrorContent', ''),
                    'InstanceId': invocation['InstanceId']
                }
                
                log_info("task", taskId, "checkCompletion", "completed", f"Command status: {status}",session_id=sessionId)
                return completionDetails
                
            except Exception as e:
                if attempt < max_retries - 1:
                    log_info("task", taskId, "checkCompletion", "retry_error", f"Error checking invocations, retrying: {str(e)}",session_id=sessionId)
                    time.sleep(retry_delay)
                    continue
                else:
                    raise
                    
    except Exception as e:
        log_error("task", taskId, "checkCompletion", "error", f"Failed to check completion: {str(e)}",session_id=sessionId)
        raise

def finish(connection, client, completionDetails, runDetails, taskId="NA", sessionId="NA"):
    try:
        log_info("task", taskId, "finish", "start", "Starting cleanup",session_id=sessionId)
        
        if completionDetails.get('status') != 'Success':
            try:
                command_id = runDetails['CommandId']
                log_info("task", taskId, "finish", "cancelling", f"Attempting to cancel command: {command_id}",session_id=sessionId)
                client.cancel_command(CommandId=command_id)
                log_info("task", taskId, "finish", "cancelled", f"Command {command_id} cancellation requested",session_id=sessionId)
            except Exception as e:
                log_error("task", taskId, "finish", "cancel_error", f"Failed to cancel command: {str(e)}",session_id=sessionId)
        
        log_info("task", taskId, "finish", "completed", "Cleanup completed",session_id=sessionId)
        
    except Exception as e:
        log_error("task", taskId, "finish", "error", f"Error during cleanup: {str(e)}",session_id=sessionId)
        raise
